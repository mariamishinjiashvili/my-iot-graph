<!DOCTYPE html>

<html lang="en">

<head>

    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>Patient Position and Duration Graph</title>

    <!-- Include Chart.js library -->

    <script src="https://cdn.jsdelivr.net/npm/chart.js"></script>

</head>

<body>

    <h1>Patient Position and Duration Graph</h1>

    <canvas id="myChart" width="800" height="400"></canvas>


    <script>

        // Function to fetch data from Oracle table using AJAX

        function fetchData() {

            // URL of the Oracle table API

            var url = 'https://fciszjhbmsy6qgt-db1n8ia.adb.us-ashburn-1.oraclecloudapps.com/ords/iot/bedlog/';


            // Fetch data using AJAX

            fetch(url)

                .then(response => response.json())

                .then(data => {

                    // Extract patient position and duration from the fetched data

                    var positions = data.items.map(item => item.PATIENT_POSITION);

                    var durations = data.items.map(item => item.PATIENT_DURATION);


                    // Call function to create chart

                    createChart(positions, durations);

                })

                .catch(error => console.error('Error fetching data:', error));

        }


        // Function to create a chart using Chart.js

        function createChart(positions, durations) {

            var ctx = document.getElementById('myChart').getContext('2d');

            var myChart = new Chart(ctx, {

                type: 'bar',

                data: {

                    labels: positions,

                    datasets: [{

                        label: 'Patient Duration',

                        data: durations,

                        backgroundColor: 'rgba(255, 99, 132, 0.2)',

                        borderColor: 'rgba(255, 99, 132, 1)',

                        borderWidth: 1

                    }]

                },

                options: {

                    scales: {

                        y: {

                            beginAtZero: true

                        }

                    }

                }

            });

        }


        // Call fetchData function to fetch data and create chart

        fetchData();

    </script>

</body>

</html>
